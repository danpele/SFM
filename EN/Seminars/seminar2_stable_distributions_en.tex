\section{Part I: Manual exercises}

\begin{frame}{Exercise 1: Verifying Stability --- The Gaussian Case}
\begin{block}{Problem}
Let $X_1, X_2$ be i.i.d.\ $\sim \mathcal{N}(0, 1)$. Show that
$a X_1 + b X_2 \stackrel{d}{=} c X + d$ for $a=3, b=4, c=5$.
\end{block}

\begin{exampleblock}{Solution}
If $X_1 \sim \mathcal{N}(0, 1)$, then $aX_1 \sim \mathcal{N}(0, a^2)$ and $bX_2 \sim \mathcal{N}(0, b^2)$.

Sum: $aX_1 + bX_2 \sim \mathcal{N}(0, a^2 + b^2) = \mathcal{N}(0, 9 + 16) = \mathcal{N}(0, 25)$.

Hence $c = \sqrt{a^2 + b^2} = \sqrt{25} = 5$ and $d = 0$.

\vspace{0.2cm}
\textbf{Conclusion:} The normal distribution is stable with $\alpha = 2$. The stability property reduces to the variance addition rule.
\end{exampleblock}
\end{frame}

\begin{frame}{Exercise 1 cont.: Cauchy Case + Comparison Table}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{Cauchy case ($\alpha = 1$)}
Let $X_1, X_2 \sim \text{Cauchy}(0,1)$ i.i.d.

$aX_1 + bX_2 \sim \text{Cauchy}(0, a+b)$

For $a=3, b=4$: $c = a + b = 7$, $d = 0$.

\vspace{0.2cm}
\textbf{Note:} For Cauchy, scales add \textit{linearly}, not quadratically!
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{General rule}
For $S(\alpha, \beta, \gamma, \delta)$:
$$c^\alpha = a^\alpha + b^\alpha$$

\vspace{0.2cm}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lccc}
\toprule
& $\alpha$ & Rule & $c$ \\
\midrule
Gaussian & 2 & $c^2 = a^2+b^2$ & 5 \\
Cauchy & 1 & $c = a+b$ & 7 \\
$S(1.5)$ & 1.5 & $c^{1.5} = a^{1.5}+b^{1.5}$ & 6.19 \\
\bottomrule
\end{tabular}
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Exercise 2: The Characteristic Function}
\begin{block}{Problem}
Compute $\varphi_X(t)$ for $X \sim S(1.5, 0.3, 1, 0)$ at $t = 1$.
\end{block}

\begin{exampleblock}{Solution}
The characteristic function of a stable distribution:
$$\log \varphi_X(t) = -|t|^\alpha \left[1 - i\beta\,\text{sgn}(t)\tan\frac{\pi\alpha}{2}\right] + i\delta t$$

For $\alpha = 1.5$, $\beta = 0.3$, $\gamma = 1$, $\delta = 0$, $t = 1$:
$$\log \varphi_X(1) = -|1|^{1.5}\left[1 - i \cdot 0.3 \cdot 1 \cdot \tan\frac{3\pi}{4}\right]$$
$$= -1 \cdot [1 - i \cdot 0.3 \cdot (-1)] = -1 - 0.3i$$

Therefore:
$$\varphi_X(1) = e^{-1 - 0.3i} = e^{-1}(\cos 0.3 - i\sin 0.3) \approx 0.351 - 0.109i$$
$$|\varphi_X(1)| = e^{-1} \approx 0.368$$
\end{exampleblock}
\end{frame}

\begin{frame}{Exercise 3: Moments and Tails}
\begin{block}{Problem}
A financial analyst models daily returns with $S(1.7, -0.2, 0.01, 0.0005)$. Which moments exist? What are the risk management implications?
\end{block}

\begin{exampleblock}{Solution}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Moment existence:}
\begin{itemize}
    \item $\E[|X|^p] < \infty$ if and only if $p < \alpha$
    \item $\alpha = 1.7 \Rightarrow$
    \begin{itemize}
        \item Mean ($p=1$): \textcolor{Forest}{exists} ($1 < 1.7$)
        \item Variance ($p=2$): \textcolor{Crimson}{does not exist} ($2 > 1.7$)
        \item Kurtosis ($p=4$): \textcolor{Crimson}{does not exist}
    \end{itemize}
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Practical implications:}
\begin{itemize}
    \item $\beta = -0.2$: negative skewness (downside risk)
    \item Gaussian VaR \textbf{underestimates} true risk
    \item Probability of a $-5\sigma$ event:
    \begin{itemize}
        \item Normal: $\approx 3 \times 10^{-7}$ (1 in 3.5M days)
        \item Stable(1.7): $\approx 0.003$ (1 in 333 days)
    \end{itemize}
    \item Sharpe ratio is not rigorously defined!
\end{itemize}
\end{column}
\end{columns}
\end{exampleblock}
\end{frame}

\section{Part II: Estimation}

\begin{frame}{McCulloch's Method --- Concept}
\begin{block}{Key idea}
Estimate parameters $(\alpha, \beta, \gamma, \delta)$ from \textbf{sample quantiles}.
\end{block}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Intermediate statistics:}
$$v_\alpha = \frac{q_{0.95} - q_{0.05}}{q_{0.75} - q_{0.25}}$$
$$v_\beta = \frac{q_{0.95} + q_{0.05} - 2q_{0.50}}{q_{0.95} - q_{0.05}}$$

\vspace{0.2cm}
\begin{itemize}
    \item $v_\alpha$ measures \textit{tail thickness}
    \item $v_\beta$ measures \textit{asymmetry}
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Reference table (selection):}

\renewcommand{\arraystretch}{1.2}
\begin{tabular}{ccc}
\toprule
$v_\alpha$ & $\alpha$ & Interpretation \\
\midrule
2.44 & 2.0 & Gaussian \\
2.62 & 1.9 & Near Gaussian \\
3.04 & 1.7 & Moderate tails \\
4.21 & 1.5 & Heavy tails \\
6.31 & 1.3 & Very heavy tails \\
\bottomrule
\end{tabular}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{McCulloch --- Worked Example}
\begin{block}{Numerical data}
From a sample of 2{,}000 observations: $q_{0.05} = -3.42$, $q_{0.25} = -0.78$, $q_{0.50} = 0.05$, $q_{0.75} = 0.82$, $q_{0.95} = 3.61$.
\end{block}

\begin{exampleblock}{Step-by-step solution}
\textbf{Step 1: Compute $v_\alpha$}
$$v_\alpha = \frac{q_{0.95} - q_{0.05}}{q_{0.75} - q_{0.25}} = \frac{3.61 - (-3.42)}{0.82 - (-0.78)} = \frac{7.03}{1.60} = 4.39$$

\textbf{Step 2: Estimate $\alpha$}
From the table: $v_\alpha = 4.39$ corresponds to $\hat{\alpha} \approx 1.48$.

\textbf{Step 3: Compute $v_\beta$}
$$v_\beta = \frac{q_{0.95} + q_{0.05} - 2q_{0.50}}{q_{0.95} - q_{0.05}} = \frac{3.61 + (-3.42) - 2 \cdot 0.05}{7.03} = \frac{0.09}{7.03} \approx 0.013$$

\textbf{Step 4: Estimate $\beta$}
$v_\beta \approx 0.013 \approx 0 \Rightarrow \hat{\beta} \approx 0$ (nearly symmetric distribution).
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Maximum Likelihood Estimation (MLE)}
\begin{block}{Procedure}
$$\hat{\theta}_{MLE} = \arg\max_{\theta} \sum_{i=1}^{n} \log f(x_i; \alpha, \beta, \gamma, \delta)$$
\end{block}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Challenges:}
\begin{itemize}
    \item PDF has no closed form (in general)
    \item Requires numerical evaluation via FFT
    \item Likelihood function is not convex
    \item Slow convergence for small $\alpha$
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Advantages:}
\begin{itemize}
    \item Asymptotically most efficient estimator
    \item Standard confidence intervals
    \item Modern implementation: \texttt{scipy.stats.levy\_stable.fit()}
\end{itemize}

\vspace{0.2cm}
\textbf{Python one-liner:}
\begin{lstlisting}
from scipy.stats import levy_stable
a, b, loc, sc = levy_stable.fit(data)
\end{lstlisting}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Koutrouvelis Regression}
\begin{block}{Two-stage method}
Uses the empirical characteristic function (ECF).
\end{block}

\textbf{Stage 1:} Estimate $\alpha$ and $\gamma$ via linear regression:
$$\log\left(-\log|\hat{\varphi}_n(t)|^2\right) = \log(2\gamma^\alpha) + \alpha \log|t|$$
Regression $Y = a + bX$ with $Y_k = \log(-\log|\hat{\varphi}_n(t_k)|^2)$ and $X_k = \log|t_k|$.

\vspace{0.3cm}
\textbf{Stage 2:} Estimate $\beta$ and $\delta$ via regression on the imaginary part:
$$\arg[\hat{\varphi}_n(t)] = \delta t - \beta\gamma^\alpha \tan\frac{\pi\alpha}{2}\,\text{sgn}(t)|t|^\alpha$$

\vspace{0.2cm}
\begin{alertblock}{Advantage}
Numerically robust, does not require PDF evaluation, fast convergence.
\end{alertblock}
\end{frame}

\section{Part IV: Python implementation}

\begin{frame}[fragile]{Simulating Stable Distributions}

\begin{lstlisting}
import numpy as np
from scipy.stats import levy_stable
import matplotlib.pyplot as plt

# Simulate stable distributions for different alpha
alphas = [0.5, 1.0, 1.5, 2.0]
x = np.linspace(-8, 8, 1000)

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 5))
colors = ['#DC3545', '#FF8C00', '#2E7D32', '#1A3A6E']
labels = [r'$\alpha=0.5$ (Levy)', r'$\alpha=1.0$ (Cauchy)',
          r'$\alpha=1.5$', r'$\alpha=2.0$ (Gaussian)']

for a, c, lab in zip(alphas, colors, labels):
    pdf = levy_stable.pdf(x, alpha=a, beta=0)
    ax1.plot(x, pdf, color=c, label=lab)       # Linear scale
    ax2.semilogy(x, pdf, color=c, label=lab)    # Log scale

ax1.set_title('PDFs (linear)')
ax2.set_title('PDFs (log scale) - tail behavior')

\begin{frame}[fragile]{Fitting Stable to S\&P 500}

import yfinance as yf
from scipy.stats import levy_stable, norm

# Download S&P 500 data
sp500 = yf.download('^GSPC', start='2000-01-01', end='2025-12-31')
returns = np.log(sp500['Close'] / sp500['Close'].shift(1)).dropna()

# Fit stable distribution
alpha, beta, loc, scale = levy_stable.fit(returns.values)
print(f"Stable fit: alpha={alpha:.3f}, beta={beta:.3f}")

# Fit normal
mu, sigma = norm.fit(returns.values)
print(f"Normal fit: mu={mu:.6f}, sigma={sigma:.6f}")

# Compare on log-scale
x = np.linspace(returns.min()*1.2, returns.max()*1.2, 1000)
plt.hist(returns, bins=150, density=True, alpha=0.4, label='Data')
plt.semilogy(x, levy_stable.pdf(x, alpha, beta, loc, scale),
             label=f'Stable (alpha={alpha:.2f})')
plt.semilogy(x, norm.pdf(x, mu, sigma), '--', label='Normal')
plt.legend(loc='upper center', bbox_to_anchor=(0.5, -0.1),
           ncol=3, frameon=False)
\end{lstlisting}
\end{frame}

\section{Quick quiz}

\begin{frame}{Quiz: 5 Questions}
\begin{enumerate}
    \item What are the bounds of the stability parameter $\alpha$?
    \begin{enumerate}
        \item[(a)] $0 < \alpha \leq 1$
        \item[(b)] $0 < \alpha \leq 2$
        \item[(c)] $-2 \leq \alpha \leq 2$
        \item[(d)] $\alpha > 0$
    \end{enumerate}

    \vspace{0.2cm}
    \item For a distribution $S(1.6, 0, 1, 0)$, which moments \textbf{exist}?
    \begin{enumerate}
        \item[(a)] Mean and variance
        \item[(b)] Only the mean
        \item[(c)] All moments
        \item[(d)] No moments
    \end{enumerate}

    \vspace{0.2cm}
    \item Which distribution is obtained at $\alpha = 1, \beta = 0$?
    \begin{enumerate}
        \item[(a)] Gaussian
        \item[(b)] Cauchy
        \item[(c)] L\'{e}vy
        \item[(d)] Exponential
    \end{enumerate}
\end{enumerate}
\end{frame}

\begin{frame}{Quiz: 5 Questions (cont.) + Answers}
\begin{enumerate}
    \setcounter{enumi}{3}
    \item In McCulloch's method, $v_\alpha = 2.44$ corresponds to:
    \begin{enumerate}
        \item[(a)] Cauchy
        \item[(b)] Gaussian
        \item[(c)] L\'{e}vy
        \item[(d)] Stable($\alpha = 1.5$)
    \end{enumerate}

    \vspace{0.2cm}
    \item Why is stable VaR $>$ Gaussian VaR at 0.1\%?
    \begin{enumerate}
        \item[(a)] The stable distribution has heavier tails
        \item[(b)] The stable distribution is more symmetric
        \item[(c)] The Gaussian has infinite variance
        \item[(d)] The parameter $\beta$ is negative
    \end{enumerate}
\end{enumerate}

\vspace{0.3cm}
\begin{exampleblock}{Answers}
\begin{enumerate}
    \item \textbf{(b)} $0 < \alpha \leq 2$
    \item \textbf{(b)} Only the mean (since $p < \alpha = 1.6$, only $p < 1.6$)
    \item \textbf{(b)} Cauchy
    \item \textbf{(b)} Gaussian ($\alpha = 2.0$)
    \item \textbf{(a)} Heavier tails of the stable distribution
\end{enumerate}
\end{exampleblock}
\end{frame}
