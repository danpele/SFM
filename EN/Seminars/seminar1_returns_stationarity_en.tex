\begin{frame}{Quick poll --- Answer}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Calculation}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item \$100 $\times$ 1.50 $=$ \$150, then \$150 $\times$ 0.50 $=$ \textcolor{Crimson}{\textbf{\$75}}
        \item You lose \textbf{25\%}, not 0\%!
    \end{itemize}
    \end{exampleblock}
    \vspace{0.3cm}
    \begin{alertblock}{Why?}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item The 50\% gain applies to \$100, but the 50\% loss applies to \$150 --- a \textit{larger base}
        \item This is \textbf{variance drag} in action: volatility \textit{destroys} wealth
        \item Formula: loss $\approx \sigma^2 / 2$ per year (we prove this in Part II)
    \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}[shrink=3]{Guided discussion: Why don't $+50\%$ and $-50\%$ cancel out?}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{You just saw: \$100 $\to$ \$150 $\to$ \$75. But what about log returns?}
    \end{alertblock}
    \vspace{0.2cm}
    \begin{columns}[T]
    \begin{column}{0.48\textwidth}
    \begin{block}{Simple returns (multiplicative)}
    \[
    (1.50)(0.50) = 0.75 \neq 1.00
    \]
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $R_1 + R_2 = +50\% + (-50\%) = 0\%$
        \item But terminal wealth: \$75, not \$100!
        \item Simple returns are \textbf{not} additive
    \end{itemize}
    \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
    \begin{exampleblock}{Log returns (additive)}
    \[
    r_{\text{up}} = \ln(1.50) = +40.55\%
    \]
    \[
    r_{\text{down}} = \ln(0.50) = -69.31\%
    \]
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Sum: $-28.77\% = \ln(0.75)$ \checkmark
        \item Log returns \textbf{correctly} reflect the loss
    \end{itemize}
    \end{exampleblock}
    \end{column}
    \end{columns}
    \vspace{0.2cm}
    \begin{alertblock}{Key message}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item The asymmetry is dramatic: you need $+100\%$ to recover from $-50\%$
        \item The higher the volatility, the more severe the loss
    \end{itemize}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\section{Part II: Variance Drag}

\begin{frame}{Variance Drag: Formula and intuition}
    \begin{cminipage}{0.95\textwidth}
    \begin{defn}[Variance drag]
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Relationship between the expected arithmetic and geometric return:
        \[
        \E[r] \approx \E[R] - \frac{1}{2}\sigma^2
        \]
        \item $\E[R]$ --- expected simple return
        \item $\E[r]$ --- expected log return
        \item $\sigma^2$ --- variance of simple returns
    \end{itemize}
    \end{defn}
    \vspace{0.2cm}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Variance drag grows \textbf{quadratically} with volatility: doubling $\sigma$ $\Rightarrow$ quadrupling the loss
        \item This is a consequence of \textbf{Jensen's inequality}: $\E[\ln(1+R)] < \ln(1+\E[R])$ since $\ln$ is concave
        \item \textbf{Implication}: volatility \textit{destroys} wealth in the long run, even if the arithmetic return is positive
    \end{itemize}
    \end{cminipage}
\end{frame}

\begin{frame}[shrink=7]{Exercises 3--5}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Exercise 3: Additivity of log returns --- Show that $r_{0,T} = \ln(P_T/P_0) = \sum_{t=1}^{T} r_t$}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item \textit{Hint}: write $\dfrac{P_T}{P_0} = \dfrac{P_1}{P_0} \cdot \dfrac{P_2}{P_1} \cdots \dfrac{P_T}{P_{T-1}}$ and apply $\ln$ to the product.
    \end{itemize}
    \end{block}
    \vspace{0.2cm}
    \begin{block}{Exercise 4: Variance drag --- Show that $\E[r] \approx \E[R] - \frac{1}{2}\sigma^2$}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Using the Taylor expansion: $\ln(1+R) \approx R - R^2/2$, apply $\E[\cdot]$ to both sides.
        \item \textit{Hint}: $\E[R^2] = \Var(R) + (\E[R])^2 \approx \sigma^2$ for $\E[R] \approx 0$.
    \end{itemize}
    \end{block}
    \vspace{0.2cm}
    \begin{exampleblock}{Exercise 5: Jensen's inequality --- Derive $\E[\ln(1+R)] \leq \ln(1 + \E[R])$}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item The function $f(x) = \ln(x)$ is concave. State Jensen's inequality and interpret: what does this imply for the relationship between the geometric and arithmetic return?
    \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\begin{frame}[shrink=7]{Exercises 3--5: Solutions}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Solution Ex.\ 3 --- Additivity}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $\dfrac{P_T}{P_0} = \displaystyle\prod_{t=1}^{T} \dfrac{P_t}{P_{t-1}} \;\;\Rightarrow\;\; \ln\!\left(\dfrac{P_T}{P_0}\right) = \displaystyle\sum_{t=1}^{T} \underbrace{\ln\!\left(\dfrac{P_t}{P_{t-1}}\right)}_{= \, r_t} \qquad \square$
    \end{itemize}
    \end{exampleblock}
    \vspace{0.2cm}
    \begin{exampleblock}{Solution Ex.\ 4 --- Variance drag}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $r = \ln(1+R) \approx R - \tfrac{R^2}{2}$. Apply $\E[\cdot]$:
        \item $\E[r] \approx \E[R] - \tfrac{1}{2}\E[R^2] = \E[R] - \tfrac{1}{2}\bigl(\sigma^2 + \mu^2\bigr) \approx \E[R] - \tfrac{1}{2}\sigma^2$ \quad (since $\mu^2 \ll \sigma^2$) $\;\square$
    \end{itemize}
    \end{exampleblock}
    \vspace{0.2cm}
    \begin{exampleblock}{Solution Ex.\ 5 --- Jensen}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Jensen's inequality: if $f$ is concave, then $\E[f(X)] \leq f(\E[X])$
        \item With $f = \ln$, $X = 1+R$: $\;\E[\ln(1+R)] \leq \ln(1+\E[R])$
        \item Interpretation: geometric return $<$ arithmetic return; the gap increases with $\sigma$ $\;\square$
    \end{itemize}
    \end{exampleblock}
    \end{cminipage}
\end{frame}

\section{Part III: Stationarity}

\begin{frame}[shrink=10]{Why do we work with returns, not prices?}
    \begin{cminipage}{0.95\textwidth}
    \begin{columns}[T]
    \begin{column}{0.48\textwidth}
    \begin{alertblock}{Prices: $P_t \sim I(1)$}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $\E[P_t]$, $\Var(P_t)$ are \textbf{not} constant
        \item Regression on prices $\Rightarrow$ \textbf{spurious regression} (Granger \& Newbold, 1974)
    \end{itemize}
    \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
    \begin{exampleblock}{Returns: $r_t \sim I(0)$}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Approximately stationary (mean $\approx 0$, variance $\approx$ const.)
        \item Statistical methods are \textbf{valid}
    \end{itemize}
    \end{exampleblock}
    \end{column}
    \end{columns}
    \vspace{0.1cm}
    \begin{defn}[Weak stationarity (covariance stationarity)]
    A process $\{X_t\}$ is weakly stationary if, $\forall\, t, h$:
    \[
    \E[X_t] = \mu, \qquad \Var(X_t) = \sigma^2 < \infty, \qquad \Cov(X_t, X_{t+h}) = \gamma(h).
    \]
    \end{defn}
    \vspace{0.1cm}
    \begin{block}{Consequence}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $P_t \sim I(1)$: non-stationary (unit root)
        \item $r_t = \Delta \ln P_t \sim I(0)$: stationary
        \item Classical methods (OLS, $t$-tests, $F$-tests) \textbf{require stationarity}; on $I(1)$ series they produce spurious results
    \end{itemize}
    \end{block}
    \end{cminipage}
\end{frame}


\begin{frame}[shrink=7]{Random walk and non-stationarity}
    \begin{cminipage}{0.95\textwidth}
    \begin{block}{Random walk: $P_t = P_{t-1} + \varepsilon_t$, \quad $\varepsilon_t \overset{iid}{\sim} (0, \sigma^2)$}
    \end{block}
    \begin{columns}[T]
    \begin{column}{0.50\textwidth}
    \begin{alertblock}{Why is $P_t$ non-stationary?}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Iterating: $P_t = P_0 + \sum_{s=1}^{t} \varepsilon_s$
        \item $\Var(P_t) = t\,\sigma^2 \;\xrightarrow{t\to\infty}\; \infty$
        \item Variance \textbf{grows linearly} with $t$
    \end{itemize}
    \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
    \begin{exampleblock}{But $r_t = \Delta \ln P_t$ is stationary}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $\E[r_t] = \mu$ (constant)
        \item $\Var(r_t) = \sigma^2$ (constant)
        \item $\Cov(r_t, r_{t+h}) = \gamma(h)$ --- depends only on $h$
    \end{itemize}
    \end{exampleblock}
    \end{column}
    \end{columns}
    \vspace{0.1cm}
    \begin{block}{Strict vs.\ weak}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item \textbf{Strict}: the joint distribution $(X_{t_1}, \ldots, X_{t_k})$ is invariant to time shifts
        \item \textbf{Weak}: only the first two moments are invariant (sufficient for OLS, ACF)
        \item Strict $\Rightarrow$ weak (if $\E[X_t^2] < \infty$), but \textbf{not} conversely
    \end{itemize}
    \end{block}
    \end{cminipage}
\end{frame}

\begin{frame}[shrink=10]{Case study: Jarque-Bera Test --- Formal rejection of normality}
    \begin{cminipage}{0.95\textwidth}
    \begin{defn}[Jarque-Bera statistic]
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $H_0$: $r_t \sim \mathcal{N}(\mu, \sigma^2)$
        \item $JB = \dfrac{n}{6}\!\left(S^2 + \dfrac{(K-3)^2}{4}\right) \;\xrightarrow{d}\; \chi^2_2$
        \item $S$ = skewness, $K$ = kurtosis, $n$ = sample size
    \end{itemize}
    \end{defn}
    \begin{exampleblock}{Computation on S\&P 500 (2000--2024, $n = 6\,287$)}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item $JB = \frac{6287}{6}\bigl((-0.39)^2 + (13.4 - 3)^2/4\bigr) = 1047.8 \times 27.19 = \mathbf{28\,488}$
    \end{itemize}
    \end{exampleblock}
    \begin{center}
    {\small\renewcommand{\arraystretch}{1.15}
    \begin{tabular}{lccccc}
    \toprule
    \textbf{Asset} & $n$ & $S$ & $K$ & \textbf{JB} & $p$\textbf{-value} \\
    \midrule
    S\&P 500 & 6\,287 & $-0.39$ & $13.4$ & 28\,488 & $< 10^{-16}$ \\
    AAPL     & 2\,516 & $-0.31$ & $8.6$  & 3\,168  & $< 10^{-16}$ \\
    Bitcoin  & 3\,287 & $-0.15$ & $7.4$  & 2\,675  & $< 10^{-16}$ \\
    \bottomrule
    \end{tabular}}
    \end{center}
    \begin{itemize}\setlength{\itemsep}{0pt}
        \item Critical value: $\chi^2_{2, 0.01} = 9.21$
        \item All $JB \gg 9.21$ $\Rightarrow$ reject $H_0$ at \textbf{any} significance level
    \end{itemize}
    \end{cminipage}
\end{frame}

\section{Quick quiz}

\begin{frame}[shrink=3]{Quick quiz --- Seminar 1 (15 minutes)}
    \begin{cminipage}{0.95\textwidth}
    \begin{alertblock}{Answer on paper (no notes)}
    \begin{enumerate}\setlength{\itemsep}{0pt}
        \item Write the \textbf{3 conditions} for weak stationarity (covariance stationarity).
        \vspace{0.08cm}
        \item Why is ACF($r_t$) $\approx 0$ but ACF($r_t^2$) $\neq 0$? What model does this motivate?
        \vspace{0.08cm}
        \item An asset has $\E[R] = 10\%$, $\sigma = 40\%$. Compute the variance drag and wealth after 20 years ($W_0 = 1$). Interpret.
        \vspace{0.08cm}
        \item An $R^2 = 0.92$ between the price of stock A and the price of stock B necessarily indicates a real relationship? Argue.
        \vspace{0.08cm}
        \item For a sample with $n = 1000$, $S = -0.5$, $K = 8$, compute the Jarque-Bera statistic and conclude.
    \end{enumerate}
    \end{alertblock}
    \end{cminipage}
\end{frame}

\begin{frame}[shrink=5]{Quick quiz --- Answers}
    \begin{cminipage}{0.95\textwidth}
    \begin{exampleblock}{Answers}
    \begin{enumerate}\setlength{\itemsep}{0pt}
        \item $\E[X_t] = \mu$; \; $\Var(X_t) = \sigma^2 < \infty$; \; $\Cov(X_t, X_{t+h}) = \gamma(h)$ --- the first two moments are constant, autocovariance depends only on the lag $h$.
        \vspace{0.05cm}
        \item Returns are uncorrelated (EMH) but \textbf{not independent}: their magnitude is autocorrelated. ACF($r_t^2$) captures \textbf{volatility clustering} $\Rightarrow$ GARCH motivation.
        \vspace{0.05cm}
        \item Drag $= \sigma^2/2 = 0.40^2/2 = 8\%$/yr. $\E[r] = 10\% - 8\% = 2\%$/yr. $W_{20} = e^{20 \times 0.02} = 1.49$. The 10\% arithmetic return ``melts'' to just 2\% geometric --- extreme volatility destroys wealth.
        \vspace{0.05cm}
        \item \textbf{No.} Both prices can be $I(1)$ with common trends (spurious regression, Granger \& Newbold 1974). Must test on returns ($R^2$ drops dramatically) or test for cointegration.
        \vspace{0.05cm}
        \item $JB = \frac{1000}{6}((-0.5)^2 + (8-3)^2/4) = 166.7 \times (0.25 + 6.25) = \mathbf{1083.3}$. $\chi^2_{2, 0.01} = 9.21$. $1083.3 \gg 9.21$ $\Rightarrow$ reject $H_0$ of normality.
    \end{enumerate}
    \end{exampleblock}
    \end{cminipage}
\end{frame}
