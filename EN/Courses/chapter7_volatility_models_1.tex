%=============================================================================
% SECTION 2: ARCH Model
%=============================================================================
\section{ARCH Model}

\begin{frame}{ARCH Model}
    \begin{itemize}
        \item Engle's (1982) ARCH($q$) specification
        \item Conditional variance: $\sigma_t^2 = \omega + \sum_{i=1}^{q} \alpha_i \varepsilon_{t-i}^2$
        \item Stationarity conditions and unconditional variance
        \item Properties: excess kurtosis and volatility clustering
    \end{itemize}
\end{frame}

%=============================================================================
% SECTION 3: GARCH Model
%=============================================================================
\section{GARCH Model}

\begin{frame}{GARCH Model}
    \begin{itemize}
        \item Bollerslev's (1986) GARCH($p,q$) generalization
        \item Conditional variance: $\sigma_t^2 = \omega + \sum_{i=1}^{q} \alpha_i \varepsilon_{t-i}^2 + \sum_{j=1}^{p} \beta_j \sigma_{t-j}^2$
        \item Persistence and the IGARCH case
        \item GARCH(1,1) as the workhorse model
    \end{itemize}
\end{frame}

%=============================================================================
% SECTION 4: Estimation
%=============================================================================
\section{Estimation}

\begin{frame}{Estimation}
    \begin{itemize}
        \item Maximum Likelihood Estimation (MLE)
        \item Log-likelihood function for GARCH models
        \item Quasi-MLE under non-Gaussian innovations
        \item Numerical optimization: BFGS, constraints
    \end{itemize}
\end{frame}

%=============================================================================
% SECTION 5: Model Selection
%=============================================================================
\section{Model Selection}

\begin{frame}{Model Selection}
    \begin{itemize}
        \item Information criteria: AIC, BIC, HQIC
        \item Likelihood ratio tests for nested models
        \item Residual diagnostics: standardized residuals, Ljung--Box test on $\hat{z}_t^2$
        \item ARCH-LM test for remaining heteroskedasticity
    \end{itemize}
\end{frame}
