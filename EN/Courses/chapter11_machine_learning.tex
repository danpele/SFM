% SECTION 2: LASSO Regression
%=============================================================================
\section{LASSO Regression}

\begin{frame}{LASSO Regression}
    \begin{itemize}
        \item $L_1$-penalized regression: $\min_{\bm{\beta}} \frac{1}{2n} \| \mathbf{y} - \mathbf{X}\bm{\beta} \|_2^2 + \lambda \|\bm{\beta}\|_1$
        \item Sparsity and automatic variable selection
        \item Cross-validation for tuning $\lambda$
        \item Comparison with Ridge regression and Elastic Net
        \item Applications: factor selection in asset pricing
    \end{itemize}
\end{frame}

%=============================================================================

\section{Random Forest}

\begin{frame}{Random Forest}
    \begin{itemize}
        \item Decision trees as base learners
        \item Bagging and random feature subsampling
        \item Out-of-bag error estimation
        \item Variable importance measures
        \item Advantages: non-linearity, robustness to outliers
    \end{itemize}
\end{frame}
